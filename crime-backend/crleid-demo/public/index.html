Crime Intelligence View
vw_Crime_Intelligence — JOINs across 6 tables with aggregated counts
-- Oracle VIEW definition SELECT c.CrimeID, c.CrimeType, c.Severity, c.Status, ps.StationName, f.FIRNumber, f.FIRStatus, COUNT(DISTINCT cs.SuspectID) AS SuspectCount, COUNT(DISTINCT v.VictimID) AS VictimCount, COUNT(DISTINCT w.WitnessID) AS WitnessCount FROM Crime c JOIN Police_Station ps ON c.StationID = ps.StationID LEFT JOIN FIR f ON c.CrimeID = f.CrimeID LEFT JOIN Crime_Suspect cs ON c.CrimeID = cs.CrimeID LEFT JOIN Victims v ON c.CrimeID = v.CrimeID LEFT JOIN Witnesses w ON c.CrimeID = w.CrimeID GROUP BY c.CrimeID, c.CrimeType, c.Severity, c.Status, ps.StationName, f.FIRNumber, f.FIRStatus;

Officer Workload View
vw_Officer_Workload — active case count per officer
SELECT o.OfficerID, o.Name, o.Designation, ps.StationName, COUNT(cf.CaseID) AS ActiveCases FROM Officers o JOIN Police_Station ps ON o.StationID = ps.StationID LEFT JOIN CaseFile cf ON o.OfficerID = cf.AssignedOfficerID AND cf.CaseStatus != 'Closed' GROUP BY o.OfficerID, o.Name, o.Designation, ps.StationName;

Audit Trigger — trg_Officer_Assignment_Audit
AFTER INSERT OR UPDATE on CaseFile automatically logs every officer assignment to Oracle
CREATE OR REPLACE TRIGGER trg_Officer_Assignment_Audit AFTER INSERT OR UPDATE OF AssignedOfficerID ON CaseFile FOR EACH ROW BEGIN INSERT INTO Officer_Assignment_Log (CaseID, OfficerID, AssignedAt, ActionType) VALUES (:NEW.CaseID, :NEW.AssignedOfficerID, SYSTIMESTAMP, CASE WHEN INSERTING THEN 'ASSIGNED' ELSE 'REASSIGNED' END); END;

Oracle window functions — RANK(), LISTAGG(), running SUM() — executed live on the database
SELECT ps.StationName, COUNT(c.CrimeID) AS TotalCrimes, RANK() OVER (ORDER BY COUNT(c.CrimeID) DESC) AS CrimeRank FROM Crime c JOIN Police_Station ps ON c.StationID = ps.StationID GROUP BY ps.StationName ORDER BY CrimeRank;
